matplotlib in ipython¶

畫出圖形存圖¶

In [3]:
from pylab import figure, plot, savefig
import numpy as np
figure(figsize=(15,10))
x = np.arange(0,180)
y = np.sin(x * np.pi / 180.0)
plot(x,y)
savefig('matplotlib.png')

In [16]:
figure(figsize=(15,10))
x = np.arange(0,180)
y = np.sin(x * np.pi / 180.0)
z = np.cos(x * np.pi / 180.0)
ylim(-1.1,1.1)
plot(x,y)
plot(x,z) #順手多畫一條cos的線。
