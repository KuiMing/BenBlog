## decimal zip
def solution(A, B):
        
    a = str(A)
    b = str(B)
    output = ""
    if len(a)<=len(b):
        length=len(a)
        rest=b[length:]
    else:
        length=len(b)
        rest=a[length:]
        
    for i in range(0,length):
        if (a[i]):
            output = output + a[i];
        if (b[i]):
            output = output + b[i];
    output+=rest
    output=int(output)
    return output

## maximum time
import itertools
from datetime import datetime
def solution(a, b, c, d):
    for t in itertools.permutations([a, b, c, d]):
        clock = ''.join(map(str, t))
        try:
            new_time = datetime.strptime(clock,'%H%M')

            if 'max_time' not in locals():
                max_time = new_time
                maxclock=clock
            elif new_time > max_time:
                max_time = new_time
                maxclock=clock
        except ValueError:
            pass

    if 'max_time' not in locals():
        return "NOT POSSIBLE"
    else:
        return maxclock[0:2] + ":" + maxclock[2:4]

## Ticket
def solution(A):
    A=sorted(A)
    single=0
    seven=0
    i=0
    while i<len(A):
        index=A[i]+6
        days=len(filter(lambda x: x>=A[i] and x<=index, A))
        if days>3:
            seven+=1
        elif days>0:
            single+=days
        else:
            single+=1
        try:
            i = A.index(filter(lambda x: x>index, A)[0])
        except:
            break
    payment = (single * 2) + (seven * 7)
    if (payment > 25):
        payment = 25
    return payment
